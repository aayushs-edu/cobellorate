# changes to the sql DB that we can incorporate to manage chat functionality

# for creating the table
CREATE TABLE messages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    senderID VARCAR(45),
    projectID VARCAR(45),
    message TEXT,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

# example of fetching a message
SELECT * FROM messages WHERE (senderID = <senderID> AND projectID = <projectID>) ORDER BY timestamp;
